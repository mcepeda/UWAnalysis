scram pro -n Wbb_slc6_472_CMSSW_5_3_14_patch1 CMSSW CMSSW_5_3_14_patch1
cd Wbb_slc6_472_CMSSW_5_3_14_patch1/src
cmsenv

eval `ssh-agent -s`
